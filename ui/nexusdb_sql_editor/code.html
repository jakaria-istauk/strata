-- Fetching active users for quarterly analysis
SELECT u.id, u.username, u.email, p.plan_type, p.status
FROM users u
JOIN subscriptions p ON u.id = p.user_id
WHERE u.active = true
AND p.status = 'active'
ORDER BY u.created_at DESC;
